Mastering Custom Animations in Flutter: Beyond the Basics

Why Animations Matter

Animations aren't just eye candy. They guide users through your app, provide feedback, and create a sense of direct manipulation. A well-animated app feels polished and professional, while a static one feels lifeless ‚Äî even if the functionality is identical.

Flutter's animation system is incredibly powerful but can be intimidating. In this guide, I'll break it down from simple implicit animations to complex custom painters, with real examples you can use in production apps.

Level 1: Implicit Animations

The easiest way to animate in Flutter. Just swap a regular widget for its "Animated" counterpart, and Flutter handles the rest:

class AnimatedCard extends StatefulWidget {
  @override
  State<AnimatedCard> createState() => _AnimatedCardState();
}

class _AnimatedCardState extends State<AnimatedCard> {
  bool _isExpanded = false;

  @override
  Widget build(BuildContext context) {
    return GestureDetector(
      onTap: () => setState(() => _isExpanded = !_isExpanded),
      child: AnimatedContainer(
        duration: const Duration(milliseconds: 400),
        curve: Curves.easeOutCubic,
        width: _isExpanded ? 300 : 200,
        height: _isExpanded ? 200 : 100,
        decoration: BoxDecoration(
          color: _isExpanded ? Colors.blue : Colors.grey[800],
          borderRadius: BorderRadius.circular(_isExpanded ? 24 : 12),
          boxShadow: [
            BoxShadow(
              color: Colors.blue.withOpacity(_isExpanded ? 0.3 : 0),
              blurRadius: _isExpanded ? 20 : 0,
              spreadRadius: _isExpanded ? 2 : 0,
            ),
          ],
        ),
        child: AnimatedOpacity(
          opacity: _isExpanded ? 1.0 : 0.7,
          duration: const Duration(milliseconds: 300),
          child: const Center(child: Text('Tap me')),
        ),
      ),
    );
  }
}

Implicit animation widgets include AnimatedContainer, AnimatedOpacity, AnimatedPadding, AnimatedPositioned, AnimatedScale, and many more. They're perfect for simple state-driven transitions.

Level 2: Explicit Animations with AnimationController

When you need more control ‚Äî repeating, reversing, or choreographing multiple properties ‚Äî use an AnimationController:

class PulseButton extends StatefulWidget {
  final VoidCallback onPressed;
  final Widget child;

  const PulseButton({required this.onPressed, required this.child});

  @override
  State<PulseButton> createState() => _PulseButtonState();
}

class _PulseButtonState extends State<PulseButton>
    with SingleTickerProviderStateMixin {
  late final AnimationController _controller;
  late final Animation<double> _scaleAnimation;
  late final Animation<double> _glowAnimation;

  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1500),
    )..repeat(reverse: true);

    _scaleAnimation = Tween<double>(begin: 1.0, end: 1.05).animate(
      CurvedAnimation(parent: _controller, curve: Curves.easeInOut),
    );

    _glowAnimation = Tween<double>(begin: 0.0, end: 0.4).animate(
      CurvedAnimation(parent: _controller, curve: Curves.easeInOut),
    );
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: _controller,
      builder: (context, child) {
        return Transform.scale(
          scale: _scaleAnimation.value,
          child: Container(
            decoration: BoxDecoration(
              borderRadius: BorderRadius.circular(16),
              boxShadow: [
                BoxShadow(
                  color: Colors.blue.withOpacity(_glowAnimation.value),
                  blurRadius: 20,
                  spreadRadius: 4,
                ),
              ],
            ),
            child: child,
          ),
        );
      },
      child: ElevatedButton(
        onPressed: widget.onPressed,
        child: widget.child,
      ),
    );
  }
}

Key concepts: SingleTickerProviderStateMixin provides the vsync for smooth 60fps animation. Always dispose the controller. Use CurvedAnimation for non-linear easing.

Level 3: Staggered Animations

For complex sequences where multiple elements animate in choreographed order:

class StaggeredList extends StatefulWidget {
  final List<String> items;
  const StaggeredList({required this.items});

  @override
  State<StaggeredList> createState() => _StaggeredListState();
}

class _StaggeredListState extends State<StaggeredList>
    with SingleTickerProviderStateMixin {
  late final AnimationController _controller;

  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      vsync: this,
      duration: Duration(milliseconds: 200 * widget.items.length + 400),
    )..forward();
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  Animation<double> _buildSlideAnimation(int index) {
    final start = index * 0.1;
    final end = start + 0.4;
    return Tween<double>(begin: 50, end: 0).animate(
      CurvedAnimation(
        parent: _controller,
        curve: Interval(start.clamp(0, 1), end.clamp(0, 1),
            curve: Curves.easeOutCubic),
      ),
    );
  }

  Animation<double> _buildFadeAnimation(int index) {
    final start = index * 0.1;
    final end = start + 0.3;
    return Tween<double>(begin: 0, end: 1).animate(
      CurvedAnimation(
        parent: _controller,
        curve: Interval(start.clamp(0, 1), end.clamp(0, 1)),
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: _controller,
      builder: (context, _) {
        return Column(
          children: List.generate(widget.items.length, (i) {
            return Transform.translate(
              offset: Offset(0, _buildSlideAnimation(i).value),
              child: Opacity(
                opacity: _buildFadeAnimation(i).value,
                child: ListTile(title: Text(widget.items[i])),
              ),
            );
          }),
        );
      },
    );
  }
}

The Interval class splits the controller's timeline into segments. Each item starts its animation slightly after the previous one, creating a cascading effect that looks beautiful with lists.

Level 4: Custom Paint Animations

For truly custom visuals ‚Äî particle effects, progress indicators, or data visualizations ‚Äî use CustomPainter:

class WaveProgressPainter extends CustomPainter {
  final double progress;
  final double wavePhase;
  final Color color;

  WaveProgressPainter({
    required this.progress,
    required this.wavePhase,
    required this.color,
  });

  @override
  void paint(Canvas canvas, Size size) {
    final paint = Paint()
      ..color = color.withOpacity(0.3)
      ..style = PaintingStyle.fill;

    final path = Path();
    final waterLevel = size.height * (1 - progress);

    path.moveTo(0, waterLevel);
    for (double x = 0; x <= size.width; x++) {
      final y = waterLevel +
          sin((x / size.width * 2 * pi) + wavePhase) * 8 +
          sin((x / size.width * 4 * pi) + wavePhase * 1.5) * 4;
      path.lineTo(x, y);
    }
    path.lineTo(size.width, size.height);
    path.lineTo(0, size.height);
    path.close();

    canvas.clipRRect(
      RRect.fromRectAndRadius(
        Rect.fromLTWH(0, 0, size.width, size.height),
        const Radius.circular(20),
      ),
    );
    canvas.drawPath(path, paint);

    // Second wave layer for depth
    paint.color = color.withOpacity(0.6);
    final path2 = Path();
    path2.moveTo(0, waterLevel);
    for (double x = 0; x <= size.width; x++) {
      final y = waterLevel +
          sin((x / size.width * 2 * pi) + wavePhase + 1) * 6;
      path2.lineTo(x, y);
    }
    path2.lineTo(size.width, size.height);
    path2.lineTo(0, size.height);
    path2.close();
    canvas.drawPath(path2, paint);
  }

  @override
  bool shouldRepaint(WaveProgressPainter old) =>
      old.progress != progress || old.wavePhase != wavePhase;
}

Level 5: Hero and Route Transitions

Hero animations create seamless transitions between screens. Customize them for unique effects:

// Source screen
Hero(
  tag: 'project-${project.id}',
  child: ClipRRect(
    borderRadius: BorderRadius.circular(16),
    child: Image.network(project.imageUrl, fit: BoxFit.cover),
  ),
)

// Destination screen
Hero(
  tag: 'project-${project.id}',
  flightShuttleBuilder: (flightContext, animation, direction,
      fromHeroContext, toHeroContext) {
    return AnimatedBuilder(
      animation: animation,
      builder: (context, _) {
        return ClipRRect(
          borderRadius: BorderRadius.circular(
            lerpDouble(16, 0, animation.value)!,
          ),
          child: toHeroContext.widget,
        );
      },
    );
  },
  child: Image.network(project.imageUrl,
      width: double.infinity, fit: BoxFit.cover),
)

Performance Tips

  • Use RepaintBoundary ‚Äî Wrap animated widgets to prevent the entire widget tree from repainting.
  • Avoid Opacity widget ‚Äî Use FadeTransition instead; Opacity creates a new layer on every frame.
  • Keep AnimatedBuilder's child ‚Äî Pass non-animated parts as the child parameter so they're not rebuilt each frame.
  • Profile with DevTools ‚Äî Check for jank in the performance overlay. Target 16ms per frame for 60fps.
  • Use addPostFrameCallback ‚Äî Start animations after the first frame renders to avoid blocking the initial build.

When to Use What

Technique Use When
ImplicitSimple state changes (expand, fade, color shift)
ExplicitRepeating, reversing, or precise control needed
StaggeredLists or multiple elements animating in sequence
CustomPaintCustom graphics, particles, unique visual effects
HeroShared element transitions between screens

Conclusion

Flutter gives you animation tools for every level of complexity. Start with implicit animations for simple transitions, graduate to explicit controllers when you need more control, and reach for CustomPainter when you need pixel-level creativity. The key is picking the right tool for the job ‚Äî don't overcomplicate simple animations, but don't shy away from custom solutions when the design calls for it.
